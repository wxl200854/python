			    	if(that.category && that.category[sort-1]['vip'] == 2 && that.category[sort-1]['unlock'] == 0){
			    		var unlockBookcoin = 0;
			    	    //计算解锁书币价格
			    	    unlockBookcoin = (parseInt(that.category[sort-1].words) / 1000) * 0.07 * 10;			//该章所需书币总额
			    	    unlockBookcoin = Math.round((unlockBookcoin * 10)) / 10;
			    	    that.sort = sort;
			    	    that.unlockCharpter();
			    	    return;
			    	}
			    }
	        	if(that.charpters_obj[sort] && that.charpters_obj[sort].content != ''){
	        		that.sort = sort;
			    	if(that.read_type == 'book'){
			    		document.documentElement.scrollTop = 0;
			    	}
			    	return;
	        	}
		        httpRequest('/api1/book/charpter', data, function (rs) {
		        	var charpter_data = rs.charpterData[0];
		        	that.charpters.push( charpter_data )
		        	that.charpters_obj[charpter_data['sort']] = charpter_data;
		        	that.sort = sort;
			    	if(that.read_type == 'book'){
			    		document.documentElement.scrollTop = 0;
			    	}
			    	if(!that.category){
						that.getCategory();
			    	}
			    	
			    	if($("#charpter_"+sort)){
	                    setTimeout(function(){
	                        $("html,body").animate({scrollTop:$("#charpter_"+sort).offset().top},1000)
	                    },500);
			    	}
			    	if(charpter_data['vip'] == 2){
			    		var unlockBookcoin = 0;
			    	    //计算解锁书币价格
			    	    unlockBookcoin = (parseInt(charpter_data.words) / 1000) * 0.07 * 10;			//该章所需书币总额
			    	    unlockBookcoin = Math.round((unlockBookcoin * 10)) / 10;
			    	    that.unlockBookcoin = unlockBookcoin;
			    	}else if(charpter_data['vip'] == 3 && that.unlockBookcoin == 0){
			    		var total = that.bookInfo.short_num;
		    	        if (that.bookInfo.short_type == 1){/* 现金单位 */
		    	            total = total * 10;
		    	        }
			    	    that.unlockBookcoin = total;
			    	}
		        });
			},
			unlockCharpter: function(){
				
				var that = this;
				var userinfo = that.userInfo;
				if(!userinfo ) return;
				var coins = that.unlockBookcoin;
				//用户vip 八折
				if (userinfo['vip'] == 1) {
					coins = Math.round(coins * 0.8 * 10) / 10;
				}
				var balance = userinfo['bookvoucher'] + userinfo['v2_bookcoin'];

				if (that.bookInfo.vip == 1) {
					//不是vip  去开通
	    			this.waitpay = 1;
					window.open('/recharge/vip');
				}
				
				if (balance < coins) {
					//去支付
	    			this.waitpay = 1;
					window.open('/recharge');
				} else if (balance > coins) {
					var sort = that.sort;
					//去解锁
					if (that.bookInfo.vip == 2) {
						httpRequest('/api1/user/unlockCharpter', {
							bid: that.bookInfo.id,
							sort: sort
						},
						function(data) {
							that.getCharpter(sort,1); //重新获取章节内容
							that.getUserInfo();//更新用户余额
							that.category[sort-1]['unlock'] = 1;
						},
						1,
						function(data) { //失败方法
							if (data.msg == '用户余额不足') {
								that.getUserInfo();//更新用户余额
							}
						});
					} else if (that.bookInfo.vip == 3) {
						httpRequest('/api1/user/unlockBook', {
							bid: that.bookInfo.id
						},
						function(data) {
							for(index in that.category){
								that.category[index]['unlock'] = 1;
							}
							that.getCharpter(sort,1); //重新获取章节内容
							that.getUserInfo();//更新用户余额
						});
					}
					//解锁成功后加入书架
					if (that.inBookShelf == 0) {
						that.addBookShelf();
					}
				}
			},
			nextCharpter: function(){
				if(this.read_type == 'screen' && this.charpters_obj[this.sort] && this.charpters_obj[this.sort].vip>0 && this.charpters_obj[this.sort].unlock==0){
					return;//如果瀑布流最后一个未解锁，不继续加载
				}
				if(this.charpters_obj[this.sort+1] && this.read_type == 'screen'){
					return;//如果瀑布流下一个已存在
				}
				
				if(this.sort==this.cnum){
					showMsg('已经是最后一章了');
					return;
				}
				this.getCharpter(this.sort+1);
			},
			lastCharpter: function(){
				if(this.sort==1) {
					showMsg('已经是第一章了');
					return;
				}
				this.getCharpter(this.sort-1);
			},
			getSetting: function(){
				var setting = $.cookie('setting');
		        if (setting){
					setting = JSON.parse(setting);

			    	this.theme=setting.theme,
			    	this.font_family=setting.font_family,

		        }

   					theme:         this.theme,		
   					font_family:   this.font_family,
   					read_type:     this.read_type,  
   					font_size:     this.font_size

	    	    $.cookie('setting', JSON.stringify(setting), {path:'/'});

	    },
